## Discord API for New DMs

Discord HTTP API provides:
```
POST /users/@me/channels
{
  "recipient_id": "USER_ID_HERE"
}
```

This creates a DM channel with the specified user. Returns a Channel object that can be converted to a Conversation.

**Serenity equivalent:** `Http::create_dm(user_id).await` or `User::create_dm_channel()`

  interface Props {
    onSend: (content: string, replyToMessageId?: string) => Promise<void>;
    disabled?: boolean;
    replyingTo?: Message | null;  // ← New prop

async fn create_dm_conversation(
    &self,
    user_id: &str,
) -> Result<Conversation, AdapterError> {
    let user_id = user_id.parse::<u64>()
        .map_err(|_| DiscordError::InvalidId("Invalid user ID".into()))?;
    
    // Create DM channel via Discord API
    let channel = self.http.create_private_channel(UserId::new(user_id))
        .await
        .map_err(|e| DiscordError::Api(e.to_string()))?;
    
    // Convert Channel to Conversation
    let conversation = Conversation {
        id: 0, // Will be set by database
        protocol_conversation_id: channel.id.to_string(),
        account_id: self.account_id.clone(),
        protocol: Protocol::Discord,
        name: channel.recipient.as_ref()
            .map(|u| u.name.clone())
            .unwrap_or_else(|| "Unknown".into()),
        participants: vec![channel.recipient.as_ref()
            .map(|u| u.name.clone())
            .unwrap_or_else(|| "Unknown".into())],
        unread_count: 0,
        last_message_at: None,
        last_message_preview: None,
    };
    
    Ok(conversation)
}
```

**3. Add command (conversations.rs):**

#[tauri::command]
pub async fn create_dm_conversation(
    account_id: String,
    user_id: String,
    state: State<'_, AppState>,
) -> Result<Conversation, String> {
    let registry = state.protocol_registry.lock().await;
    let adapter = registry
        .get_adapter(&account_id)
        .await
        .map_err(|e| e.to_string())?;
    
    let conversation = adapter
        .create_dm_conversation(&user_id)
        .await
        .map_err(|e| e.to_string())?;
    
    // Save to database
    let db = state.db.lock().await;
    let conversation_id = sqlx::query_scalar::<_, i64>(
        "INSERT INTO conversations 
         (protocol_conversation_id, account_id, protocol, name, participants, unread_count, last_message_at) 
         VALUES (?1, ?2, ?3, ?4, ?5, 0, NULL)
         RETURNING id"
    )
    .bind(&conversation.protocol_conversation_id)
    .bind(&conversation.account_id)
    .bind(format!("{:?}", conversation.protocol))
    .bind(&conversation.name)
    .bind(serde_json::to_string(&conversation.participants).unwrap())
    .fetch_one(&*db)
    .await
    .map_err(|e| e.to_string())?;
    
    let mut result = conversation;
    result.id = conversation_id;
    
    // Emit event for frontend
    state.app_handle.emit(
        "protocol:event",
        ProtocolEvent::ConversationUpdated { 
            account_id: account_id.clone(), 
            conversation: result.clone() 
        }
    ).map_err(|e| e.to_string())?;
    
    Ok(result)
}
```
